# 1 Выведите список всех файлов в папке html
ls ./var/www/html - Команды просмотр каталага с путем до папки в этом каталоге

# 2 Найдите в файле access.log все упоминания строки "404"
grep 404 -n /var/www/logs/access.log #- Утилита grep поиска в файле по шаблону c индексом -n показывает в каких строках были найдены упоминания

# 3 На основании файла 404.html сделайте файл 403.html заменив в его содержимом все упоминания 404 на 403
cd /var/www/html # заходим в папку
sed 's/404/403/g' 404.html > 403.html  #- используем скрипт sed для поиска и замены в тексте по шаблону и перенаправление вывода в новый файл 

# 4 Модифицируйте права доступа и владельца файла index.html следующим образом: Текущий пользователь должен иметь права на чтение/запись 
# Группа www-data на чтение Все остальные не имеют доступа к файлу
 groupadd www-data # создание группы www-data
 chown :www-data /var/www/html/index.html - # назначить группу доступа www-data на файл
 chmod u=rw,g=r,o= /var/www/html/index.html # изменяем права u - пользователь (чтение и запись) , q- группа (чтение), o - все другие запрет 

# 5 Сделайте символическую ссылку на файл access.log в папке html.
 ln -s /var/www/logs/access.log /var/www/html/link-access # команда ln c опцией s - создать символическую ссылку на файл var/www/logs/access.log и 
				# помещает ее в каталог /var/www/html

# 6 Выведите на экран все файлы в папке var
ls -R  /var #  команда ls c опцией R позволяет рекурсивно отображать содержимое поддиректорий;
